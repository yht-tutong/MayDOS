calc                打开计算器
cat %GREEN<filename>%WHITE      查看文件内容
cd %GREEN<to>%WHITE             更改工作目录
cls                 清屏
cp %GREEN<from>%WHITE %GREEN<to>%WHITE      复制文件
echo %GREEN<message>%WHITE      显示消息
exit                退出MayDOS
ls %GREEN<filename>%WHITE       列出该目录下的文件
man %GREEN<command>%WHITE       查看每个命令的详细信息
md %GREEN<dirname>%WHITE        创建一个新的文件夹
minesweeper         打开扫雷程序
mv %GREEN<from>%WHITE %GREEN<to>%WHITE      移动文件
notepad             打开记事本程序
register            注册用户
reboot              重启系统
rename %GREEN<old>%WHITE %GREEN<new>%WHITE  重命名文件
sh %GREEN<file>%WHITE           运行 Shell 文件（.sh）
shut                关机（可不是闹着玩的）
su %GREEN<username>%WHITE       切换用户名
sudo %GREEN<command>%WHITE      使用 root 用户执行一次命令
sysver              查看系统版本
touch %GREEN<filename>%WHITE    创建一个新的文件
usebook             获取帮助
write %GREEN<file>%WHITE %GREEN<text>%WHITE 写入文件某些文本
